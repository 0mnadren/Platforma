	CREATE TABLE IF NOT EXISTS naziv_tabele (
		naziv_kolone TIP OSOBINA ...,
		...,
		PRIMARY KEY(naziv_kolone, ...),
		FOREIGN KEY(naziv_kolone) REFERENCES naziv_tabele(naziv_kolone)
			ON UPDATE AKCIJA
			ON DELETE AKCIJA,
		...
	)
	- TIP (INTEGER, VARCHAR(50), CHAR(6), ...)
	- OSOBINA (NOT NULL, UNIQUE, AUTOINCREMENT, DEFAULT, CHECK)
	- AKCIJA (NO ACTION, RESTRICT, SET NULL, SET DEFAULT, CASCADE)
	
	DROP TABLE IF EXISTS naziv_tabele
	
	CREATE VIEW IF NOT EXISTS naziv_pogleda (naziv_kolone, ...)
	AS select_naredba

	DROP VIEW IF EXISTS naziv_pogleda

	INSERT INTO naziv_tabele (naziv_kolone, ...)
	VALUES (vrednost, ...), ...
	
	UPDATE naziv_tabele
	SET naziv_kolone = vrednost, ...
	WHERE uslov

	DELETE FROM naziv_tabele
	WHERE uslov
